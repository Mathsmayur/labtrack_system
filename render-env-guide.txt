# Copy these into the "Environment Variables" section of your Render Web Service

# Render provides 'Internal Database URL' when you create the Postgres Database.
# It looks like: postgres://user:password@hostname/dbname
# Since Spring Boot's JDBC needs 'jdbc:postgresql://' instead of 'postgres://', format it like below:

SPRING_DATASOURCE_URL=jdbc:postgresql://[YOUR_RENDER_DB_HOST]:5432/[YOUR_RENDER_DB_NAME]
SPRING_DATASOURCE_USERNAME=[YOUR_DB_USERNAME]
SPRING_DATASOURCE_PASSWORD=[YOUR_DB_PASSWORD]

# Important: Don't forget your JWT secret
JWT_SECRET=MySecretKeyForJWTTokenGeneration12345678901234567890

# Optional: Ensure it binds to the port Render expects (often 10000 or 8080)
# Render automatically injects the PORT environment variable.
SERVER_PORT=${PORT:-8080}
